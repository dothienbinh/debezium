check binhlog db (on,row,!=0)
  SHOW VARIABLES LIKE 'log_bin';
  SHOW VARIABLES LIKE 'binlog_format';
  SHOW VARIABLES LIKE 'server_id';

create user debezium db
  CREATE USER 'debezium'@'%' IDENTIFIED BY '123456';

  GRANT SELECT, RELOAD, SHOW DATABASES, REPLICATION SLAVE, REPLICATION CLIENT
  ON *.* TO 'debezium'@'%';

  FLUSH PRIVILEGES;

create connector:
curl -X POST http://localhost:8083/connectors \
    -H "Content-Type: application/json" \
    -d '{
      "name": "mysql-connector",
      "config": {
        "connector.class": "io.debezium.connector.mysql.MySqlConnector",
        "tasks.max": "1",

        "database.hostname": "mysql_container",
        "database.port": "3306",
        "database.user": "debezium",
        "database.password": "123456",
        "database.server.id": "184054",
        "database.server.name": "mysql-server-1",

        "database.include.list": "test",

        "topic.prefix": "cdc",

        "schema.history.internal.kafka.bootstrap.servers": "${KAFKA_BOOTSTRAP_SERVERS}:9092",
        "schema.history.internal.kafka.topic": "schema-changes.mysql"
      }
    }'

logs kafka:

  docker exec -it kafka bash
  kafka-console-consumer \
  --bootstrap-server localhost:9092 \
  --topic cdc.test.users \
  --from-beginning
